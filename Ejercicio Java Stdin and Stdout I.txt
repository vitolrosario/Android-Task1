import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int a = scan.nextInt();
        String b = scan.next();
        System.out.println(a);
        System.out.println(b);
        a = scan.nextInt();
        System.out.println(a);
        scan.close();
    }
}
